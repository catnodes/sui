54/// A broadcast receiver used by subscription resolvers to receive processed checkpoints.
55/// Stored in the GraphQL context; each subscriber calls `resubscribe()` to get its own
56/// receiver. Using a Receiver (not Sender) ensures that when the stream task drops its
57/// Sender, all subscribers receive `RecvError::Closed`.
58pub(crate) type CheckpointBroadcaster = broadcast::Receiver<Arc<ProcessedCheckpoint>>;
59
60/// Field mask requesting checkpoint-level and transaction-level fields needed by GraphQL resolvers.
61fn checkpoint_field_mask() -> FieldMask {
62    FieldMask::from_paths([
63        ProtoCheckpoint::path_builder().sequence_number(),
64        ProtoCheckpoint::path_builder().summary().bcs().value(),
65        ProtoCheckpoint::path_builder().signature().finish(),
66        ProtoCheckpoint::path_builder().contents().bcs().value(),
67        ProtoCheckpoint::path_builder()
68            .transactions()
69            .transaction()
70            .finish(),
71        ProtoCheckpoint::path_builder()
72            .transactions()
73            .effects()
74            .finish(),
75        ProtoCheckpoint::path_builder()
76            .transactions()
77            .events()
78            .bcs()
79            .value(),
80        ProtoCheckpoint::path_builder()
81            .transactions()
82            .signatures()
83            .bcs()
84            .value(),
85        ProtoCheckpoint::path_builder()
86            .transactions()
87            .balance_changes()
88            .finish(),
89        // Checkpoint-level objects (deduplicated across transactions).
90        // Per-transaction `transactions.objects` is not populated in streamed checkpoints.
91        ProtoCheckpoint::path_builder()
92            .objects()
93            .objects()
94            .bcs()
95            .value(),
96    ])
97}

99/// Background service that connects to a fullnode's gRPC SubscribeCheckpoints endpoint,
100/// processes incoming checkpoints, and broadcasts them to subscription resolvers.
101pub(crate) struct CheckpointStreamTask {
102    uri: Uri,
103    sender: broadcast::Sender<Arc<ProcessedCheckpoint>>,
104    broadcaster: CheckpointBroadcaster,
105    streaming_packages: Arc<StreamingPackageStore>,
106    package_eviction_tx: UnboundedSender<(u64, Vec<AccountAddress>)>,
107    readiness: Arc<SubscriptionReadiness>,
108}
109
110impl CheckpointStreamTask {
111    pub(crate) fn new(
112        uri: Uri,
113        config: &SubscriptionConfig,
114        streaming_packages: Arc<StreamingPackageStore>,
115        package_eviction_tx: UnboundedSender<(u64, Vec<AccountAddress>)>,
116        readiness: Arc<SubscriptionReadiness>,
117    ) -> Self {
118        let (sender, broadcaster) = broadcast::channel(config.broadcast_buffer);
119        Self {
120            uri,
121            sender,
122            broadcaster,
123            streaming_packages,
124            package_eviction_tx,
125            readiness,
126        }
127    }
128
129    pub(crate) fn broadcaster(&self) -> CheckpointBroadcaster {
130        self.broadcaster.resubscribe()
131    }
132
133    /// Connect to the fullnode's gRPC SubscribeCheckpoints endpoint.
134    async fn connect(&self) -> anyhow::Result<Streaming<SubscribeCheckpointsResponse>> {
135        let endpoint = Endpoint::from(self.uri.clone()).connect_timeout(CONNECTION_TIMEOUT);
136
137        let mut client = SubscriptionServiceClient::connect(endpoint)
138            .await
139            .context("Failed to connect to checkpoint stream")?
140            .max_decoding_message_size(MAX_GRPC_MESSAGE_SIZE_BYTES);
141
142        let mut request = SubscribeCheckpointsRequest::default();
143        request.read_mask = Some(checkpoint_field_mask());
144
145        let stream = client
146            .subscribe_checkpoints(request)
147            .await
148            .context("Failed to subscribe to checkpoint stream")?
149            .into_inner();
150
151        Ok(stream)
152    }
153
154    // TODO(Phase 2): Connection and stream errors currently terminate the task.
155    // Proper error handling will be addressed alongside backfilling.
156    pub(crate) fn run(self) -> Service {
157        Service::new().spawn_aborting(async move {
158            info!("Connecting to checkpoint stream at {}...", self.uri);
159            let mut stream = self.connect().await?;
160            info!("Connected to checkpoint stream at {}", self.uri);
161
162            let mut first_checkpoint_recorded = false;
163            while let Some(result) = stream.next().await {
164                let response = result.context("Checkpoint stream error")?;
165                if let Some(checkpoint) = response.checkpoint {
166                    let sequence_number = checkpoint
167                        .sequence_number
168                        .context("Checkpoint without sequence_number")?;
169                    if !first_checkpoint_recorded {
170                        self.readiness.record_first_checkpoint(sequence_number);
171                        first_checkpoint_recorded = true;
172                    }
173                    let packages = extract_packages(&checkpoint);
174                    if !packages.is_empty() {
175                        self.streaming_packages
176                            .index_packages(sequence_number, &packages);
177                        let ids = packages.iter().map(|p| p.storage_id()).collect();
178                        // Send errors only if the eviction task has exited — at that
179                        // point nothing will drain the store, but we keep serving.
180                        let _ = self.package_eviction_tx.send((sequence_number, ids));
181                    }
182                    let processed = process_checkpoint(checkpoint)?;
183                    // Ignore send errors — no active subscribers is a normal state
184                    // (e.g., no clients have connected yet). The checkpoint is simply dropped.
185                    let _ = self.sender.send(Arc::new(processed));
186                }
187            }
188
189            Ok(())
190        })
191    }
192}

194fn process_checkpoint(checkpoint: ProtoCheckpoint) -> anyhow::Result<ProcessedCheckpoint> {
195    let sequence_number = checkpoint
196        .sequence_number
197        .context("Checkpoint without sequence_number")?;
198
199    let summary: CheckpointSummary = checkpoint
200        .summary
201        .as_ref()
202        .and_then(|s| s.bcs.as_ref())
203        .context("Missing summary.bcs")?
204        .deserialize()
205        .context("Failed to deserialize checkpoint summary")?;
206
207    let contents: CheckpointContents = checkpoint
208        .contents
209        .as_ref()
210        .and_then(|c| c.bcs.as_ref())
211        .context("Missing contents.bcs")?
212        .deserialize()
213        .context("Failed to deserialize checkpoint contents")?;
214
215    let signature: AuthorityStrongQuorumSignInfo = {
216        let sdk_sig = ValidatorAggregatedSignature::try_from(
217            checkpoint.signature.as_ref().context("Missing signature")?,
218        )
219        .context("Failed to parse checkpoint signature")?;
220        AuthorityStrongQuorumSignInfo::from(sdk_sig)
221    };
222
223    let timestamp_ms = summary.timestamp_ms;
224    let cp_sequence_number = sequence_number;
225    let tx_lo = summary.network_total_transactions - checkpoint.transactions.len() as u64;
226    let checkpoint_objects = deserialize_checkpoint_objects(&checkpoint)?;
227    let transactions = checkpoint
228        .transactions
229        .iter()
230        .enumerate()
231        .map(|(i, proto)| {
232            process_transaction(
233                proto,
234                &checkpoint_objects,
235                timestamp_ms,
236                cp_sequence_number,
237                tx_lo + i as u64,
238            )
239        })
240        .collect::<anyhow::Result<Vec<_>>>()?;
241
242    Ok(ProcessedCheckpoint {
243        sequence_number,
244        summary,
245        contents,
246        signature,
247        transactions,
248    })
249}

251fn process_transaction(
252    proto: &ProtoExecutedTransaction,
253    checkpoint_objects: &BTreeMap<(ObjectID, SequenceNumber), NativeObject>,
254    timestamp_ms: u64,
255    cp_sequence_number: u64,
256    tx_sequence_number: u64,
257) -> anyhow::Result<ProcessedTransaction> {
258    let transaction_data: TransactionData = proto
259        .transaction
260        .as_ref()
261        .and_then(|t| t.bcs.as_ref())
262        .context("Missing transaction.bcs")?
263        .deserialize()
264        .context("Failed to deserialize transaction data")?;
265
266    let effects: TransactionEffects = proto
267        .effects
268        .as_ref()
269        .and_then(|e| e.bcs.as_ref())
270        .context("Missing effects.bcs")?
271        .deserialize()
272        .context("Failed to deserialize transaction effects")?;
273
274    let events = proto
275        .events
276        .as_ref()
277        .and_then(|e| e.bcs.as_ref())
278        .map(|bcs| {
279            bcs.deserialize()
280                .context("Failed to deserialize transaction events")
281        })
282        .transpose()?
283        .map(|e: TransactionEvents| e.data);
284
285    let signatures: Vec<GenericSignature> = proto
286        .signatures
287        .iter()
288        .map(|sig| {
289            let bytes = sig
290                .bcs
291                .as_ref()
292                .context("Missing signature bcs")?
293                .value
294                .as_deref()
295                .unwrap_or(&[]);
296            GenericSignature::from_bytes(bytes)
297                .map_err(|e| anyhow::anyhow!("Failed to parse user signature: {e}"))
298        })
299        .collect::<anyhow::Result<_>>()?;
300
301    let digest = *effects.transaction_digest();
302
303    let execution_objects = build_execution_objects(checkpoint_objects, proto)?;
304
305    let contents = NativeTransactionContents::ExecutedTransaction(
306        sui_indexer_alt_reader::kv_loader::ExecutedTransactionData {
307            effects: Box::new(effects),
308            events,
309            transaction_data: Box::new(transaction_data),
310            signatures,
311            balance_changes: proto.balance_changes.clone(),
312            proto_effects: proto.effects.clone(),
313            proto_transaction: proto.transaction.clone(),
314            timestamp_ms: Some(timestamp_ms),
315            cp_sequence_number: Some(cp_sequence_number),
316        },
317    );
318
319    Ok(ProcessedTransaction {
320        tx_sequence_number,
321        digest,
322        contents,
323        execution_objects,
324    })
325}

327/// Extract package objects from a streamed checkpoint. Returned packages will be
328/// inserted into the streaming index and queued for eventual eviction.
329fn extract_packages(checkpoint: &ProtoCheckpoint) -> Vec<Arc<Package>> {
330    let Some(object_set) = &checkpoint.objects else {
331        return Vec::new();
332    };
333    let mut packages = Vec::new();
334    for obj in &object_set.objects {
335        let Some(bcs) = &obj.bcs else { continue };
336        let Ok(native_obj) = bcs.deserialize::<NativeObject>() else {
337            continue;
338        };
339        let Some(move_package) = native_obj.data.try_as_package() else {
340            continue;
341        };
342        let Ok(package) = Package::read_from_package(move_package) else {
343            continue;
344        };
345        packages.push(Arc::new(package));
346    }
347    packages
348}

350/// Deserialize all objects from the checkpoint-level ObjectSet.
351fn deserialize_checkpoint_objects(
352    checkpoint: &ProtoCheckpoint,
353) -> anyhow::Result<BTreeMap<(ObjectID, SequenceNumber), NativeObject>> {
354    let mut map = BTreeMap::new();
355    if let Some(object_set) = &checkpoint.objects {
356        for obj in &object_set.objects {
357            if let Some(bcs) = &obj.bcs {
358                let native_obj: NativeObject = bcs
359                    .deserialize()
360                    .context("Failed to deserialize checkpoint object BCS")?;
361                map.insert((native_obj.id(), native_obj.version()), native_obj);
362            }
363        }
364    }
365    Ok(map)
366}

368/// Build an ExecutionObjectMap for a single transaction by filtering checkpoint-level objects
369/// using the transaction's `changed_objects` from effects.
370///
371/// Includes both input objects (previous version) and output objects (new version) so that
372/// both `inputState` and `outputState` can resolve from streaming.
373/// Objects with `DoesNotExist` output state become tombstones (None).
374fn build_execution_objects(
375    checkpoint_objects: &BTreeMap<(ObjectID, SequenceNumber), NativeObject>,
376    proto_tx: &ProtoExecutedTransaction,
377) -> anyhow::Result<ExecutionObjectMap> {
378    let mut map = BTreeMap::new();
379
380    if let Some(effects) = &proto_tx.effects {
381        let lamport_version = SequenceNumber::from_u64(
382            effects
383                .lamport_version
384                .context("Effects should have lamport_version")?,
385        );
386
387        for changed_obj in &effects.changed_objects {
388            let object_id: ObjectID = changed_obj
389                .object_id
390                .as_ref()
391                .and_then(|id| id.parse().ok())
392                .context("ChangedObject should have valid object_id")?;
393
394            // Input object (previous version, before the transaction).
395            if let Some(input_version) = changed_obj.input_version {
396                let input_version = SequenceNumber::from_u64(input_version);
397                if let Some(obj) = checkpoint_objects.get(&(object_id, input_version)) {
398                    map.insert((object_id, input_version), Some(obj.clone()));
399                }
400            }
401
402            // Output object (new version, after the transaction).
403            if changed_obj.output_state() == OutputObjectState::DoesNotExist {
404                map.insert((object_id, lamport_version), None);
405            } else if let Some(obj) = checkpoint_objects.get(&(object_id, lamport_version)) {
406                map.insert((object_id, lamport_version), Some(obj.clone()));
407            }
408        }
409    }
410
411    Ok(Arc::new(map))

4//! GQL Queries
5//! Interface to the rpc for the gql schema defined in `crates/sui-indexer-alt-graphql/schema.graphql`.
6//! Built in modules for epochs, transactions, objects, and checkpoints.
7//! No GQL type escapes this module. From here we return structures defined in this crate
8//! or bcs encoded data of runtime structures.
9//!
10//! Most of these query types are generated by cynic using the schema through the web UI:
11//! `https://generator.cynic-rs.dev/`

79    pub(crate) async fn query(
80        digest: String,
81        client: &GraphQLClient,
82    ) -> Result<Option<TransactionInfo>, Error> {
83        let query = Query::build(TransactionDataArgs {
84            digest: digest.clone(),
85        });
86        let response = client
87            .run_query(&query)
88            .await
89            .context("Failed to run transaction query")?;
90
91        let Some(transaction) = response.data.and_then(|txn| txn.transaction) else {
92            return Ok(None);
93        };
94
95        let txn_data: TransactionData = bcs::from_bytes(
96            &FastCryptoBase64::decode(
97                &transaction
98                    .transaction_bcs
99                    .ok_or_else(|| {
100                        anyhow!(format!(
101                            "Transaction data not available (None) for digest: {}",
102                            digest
103                        ),)
104                    })?
105                    .0,
106            )
107            .context(format!(
108                "Transaction data does not decode for digest: {}",
109                digest
110            ))?,
111        )
112        .context(format!(
113            "Cannot deserialize transaction data for digest {}",
114            digest
115        ))?;
116
117        let signatures: Vec<GenericSignature> = transaction
118            .signatures
119            .into_iter()
120            .map(|sig| {
121                let encoded = sig
122                    .signature_bytes
123                    .ok_or_else(|| anyhow!("Missing signatureBytes for transaction {}", digest))?;
124                let bytes = FastCryptoBase64::decode(&encoded.0).context(format!(
125                    "User signature does not decode for digest: {}",
126                    digest
127                ))?;
128                GenericSignature::from_bytes(&bytes).map_err(|e| {
129                    anyhow!("User signature parse failed for digest {}: {}", digest, e)
130                })
131            })
132            .collect::<Result<_, _>>()?;
133
134        let effect_frag = transaction
135            .effects
136            .ok_or_else(|| anyhow!("Missing effects in transaction data response"))?;
137        let effects: sui_types::effects::TransactionEffects = bcs::from_bytes(
138            &FastCryptoBase64::decode(
139                &effect_frag
140                    .effects_bcs
141                    .ok_or_else(|| anyhow!("Missing effects bcs in transaction data response"))?
142                    .0,
143            )
144            .context(format!(
145                "Transaction effects do not decode for digest: {}",
146                digest
147            ))?,
148        )
149        .context(format!(
150            "Cannot deserialize transaction effects for digest {}",
151            digest
152        ))?;
153
154        let checkpoint = effect_frag
155            .checkpoint
156            .ok_or_else(|| anyhow!("Missing checkpoint in transaction data response"))?
157            .sequence_number;
158
159        let transaction = VerifiedTransaction::new_unchecked(
160            SuiTransaction::from_generic_sig_data(txn_data, signatures),
161        );
162
163        Ok(Some(TransactionInfo {
164            transaction,
165            effects,
166            checkpoint,
167        }))
168    }

280    /// Fetch all events for a transaction by paginating through the GraphQL
281    /// events connection. Returns `None` if the transaction doesn't exist.
282    /// Returns an empty `TransactionEvents` if the transaction has no events.
283    pub(crate) async fn query(
284        digest: &str,
285        client: &GraphQLClient,
286    ) -> Result<Option<TransactionEvents>, Error> {
287        let mut all_events: Vec<Event> = Vec::new();
288        let mut cursor: Option<String> = None;
289
290        loop {
291            let operation = Query::build(EventsArgs {
292                digest: digest.to_owned(),
293                first: Some(PAGE_SIZE),
294                after: cursor,
295            });
296            let response = client
297                .run_query(&operation)
298                .await
299                .context("Failed to run events query")?;
300
301            let Some(connection) = response
302                .data
303                .and_then(|q| q.transaction)
304                .and_then(|tx| tx.effects)
305                .and_then(|fx| fx.events)
306            else {
307                if all_events.is_empty() {
308                    return Ok(Some(TransactionEvents { data: vec![] }));
309                }
310                break;
311            };
312
313            for node in connection.nodes {
314                let Some(bcs_str) = node.event_bcs else {
315                    continue;
316                };
317                let bytes = FastCryptoBase64::decode(&bcs_str.0)
318                    .context(format!("Event BCS does not decode for tx {}", digest))?;
319                let event: Event = bcs::from_bytes(&bytes)
320                    .context(format!("Cannot deserialize event for tx {}", digest))?;
321                all_events.push(event);
322            }
323
324            if !connection.page_info.has_next_page {
325                break;
326            }
327            cursor = connection.page_info.end_cursor;
328        }
329
330        Ok(Some(TransactionEvents { data: all_events }))
331    }

416    // Maximum number of keys to query in a single request.
417    // REVIEW: not clear how this translate to the 5000B limit, so
418    // we are picking a "random" and conservative number.
419    const MAX_KEYS_SIZE: usize = 30;
420
421    /// Query objects by keys and return a vector of the same length with the results, which contain
422    /// an Some<(Object, Version)> tuple if the object is found, or None if not found.
423    pub(crate) async fn query(
424        keys: &[GqlObjectKey],
425        data_store: &GraphQLClient,
426    ) -> Result<Vec<Option<(Object, u64)>>, Error> {
427        let mut results: Vec<Option<Option<(Object, u64)>>> = vec![None; keys.len()];
428        let mut standard_indices: Vec<usize> = Vec::with_capacity(keys.len());
429        let mut standard_keys: Vec<ObjectKey> = Vec::with_capacity(keys.len());
430
431        for (idx, key) in keys.iter().cloned().enumerate() {
432            match key.version_query {
433                VersionQuery::VersionAtCheckpoint {
434                    version,
435                    checkpoint,
436                } => {
437                    results[idx] = Some(
438                        query_version_at_checkpoint(key.object_id, version, checkpoint, data_store)
439                            .await?,
440                    );
441                }
442                _ => {
443                    standard_indices.push(idx);
444                    standard_keys.push(ObjectKey::from(key));
445                }
446            }
447        }
448
449        let mut keys = standard_keys;
450        let mut key_chunks = vec![];
451        while !keys.is_empty() {
452            let chunk: Vec<_> = keys.drain(..MAX_KEYS_SIZE.min(keys.len())).collect();
453            key_chunks.push(chunk);
454        }
455
456        let mut standard_results = vec![];
457
458        for keys in key_chunks {
459            let query: cynic::Operation<MultiGetObjectsQuery, MultiGetObjectsVars> =
460                MultiGetObjectsQuery::build(MultiGetObjectsVars { keys });
461            let response = data_store.run_query(&query).await?;
462
463            let list = if let Some(data) = response.data {
464                data.multi_get_objects
465            } else {
466                return Err(anyhow!(
467                    "Missing data in transaction query response. Errors: {:?}",
468                    response.errors,
469                ));
470            };
471
472            standard_results.extend(
473                list.into_iter()
474                    .map(decode_object_fragment)
475                    .collect::<Result<Vec<_>, _>>()?,
476            );
477        }
478
479        for (idx, object) in standard_indices.into_iter().zip_eq(standard_results) {
480            results[idx] = Some(object);
481        }
482
483        Ok(results
484            .into_iter()
485            .map(|result| result.expect("every query key should produce a result"))
486            .collect())
487    }

489    async fn query_version_at_checkpoint(
490        object_id: sui_types::base_types::ObjectID,
491        version: u64,
492        checkpoint: u64,
493        data_store: &GraphQLClient,
494    ) -> Result<Option<(Object, u64)>, Error> {
495        let query = VersionAtCheckpointQuery::build(VersionAtCheckpointVars {
496            sequence_number: Some(checkpoint),
497            address: SuiAddress(object_id.to_string()),
498            version: Some(version),
499        });
500        let response = data_store.run_query(&query).await?;
501        let checkpoint = response
502            .data
503            .and_then(|data| data.checkpoint)
504            .ok_or_else(|| anyhow!("Missing checkpoint in object query response"))?;
505        let scoped_query = checkpoint
506            .query
507            .ok_or_else(|| anyhow!("Missing checkpoint-scoped query in object response"))?;
508
509        decode_object_fragment(scoped_query.object)
510    }
511
512    fn decode_object_fragment(
513        frag: Option<ObjectFragment>,
514    ) -> Result<Option<(Object, u64)>, Error> {
515        match frag {
516            Some(frag) => {
517                let b64 = frag
518                    .object_bcs
519                    .ok_or_else(|| anyhow!("Object bcs is None for object"))?
520                    .0;
521                let bytes = FastCryptoBase64::decode(&b64)?;
522                let obj: Object = bcs::from_bytes(&bytes)?;
523                let version = frag
524                    .version
525                    .ok_or_else(|| anyhow!("Object version is None for object"))?;
526                Ok(Some((obj, version)))
527            }
528            None => Ok(None),
529        }
530    }

534            ObjectKey {

536                version: match key.version_query {
537                    VersionQuery::Version(v) => Some(v),

547                },
548                at_checkpoint: match key.version_query {

550                    VersionQuery::VersionAtCheckpoint {
551                        version,
552                        checkpoint,
553                    } => Some(checkpoint),
554                    _ => None,
555                },
556            }
557        }

685    /// Fetch a checkpoint by sequence number (or the latest if `None`), returning
686    /// both the verified summary envelope and the decoded checkpoint contents.
687    pub(crate) async fn query(
688        sequence_number: Option<u64>,
689        data_store: &GraphQLClient,
690    ) -> Result<Option<(VerifiedCheckpoint, CheckpointContents)>, Error> {
691        let query = Query::build(CheckpointArgs { sequence_number });
692        let response = data_store.run_query(&query).await?;
693        let Some(checkpoint) = response.data.and_then(|data| data.checkpoint) else {
694            return Ok(None);
695        };
696        Ok(Some(decode_checkpoint(checkpoint)?))
697    }
698
699    fn decode_checkpoint(
700        checkpoint: Checkpoint,
701    ) -> Result<(VerifiedCheckpoint, CheckpointContents), Error> {
702        let contents: CheckpointContents =
703            decode_bcs(checkpoint.content_bcs, "checkpoint contents")?;
704        let summary: CheckpointSummary = decode_bcs(checkpoint.summary_bcs, "checkpoint summary")?;
705        let Some(validator_signatures) = checkpoint.validator_signatures else {
706            return Err(anyhow!(
707                "Missing validator signatures in checkpoint response"
708            ));
709        };
710
711        let signature_bytes = FastCryptoBase64::decode(
712            &validator_signatures
713                .signature
714                .ok_or_else(|| anyhow!("Missing aggregated checkpoint signature"))?
715                .0,
716        )
717        .context("checkpoint signature does not decode")?;
718        let signature = AggregateAuthoritySignature::from_bytes(&signature_bytes)
719            .context("cannot deserialize aggregated checkpoint signature")?;
720        let signers_map = validator_signatures
721            .signers_map
722            .into_iter()
723            .map(|signer| {
724                u32::try_from(signer)
725                    .map_err(|_| anyhow!("negative signer index in checkpoint signature"))
726            })
727            .collect::<Result<RoaringBitmap, Error>>()?;
728
729        let certified = CertifiedCheckpointSummary::new_from_data_and_sig(
730            summary.clone(),
731            AuthorityStrongQuorumSignInfo {
732                epoch: summary.epoch,
733                signature,
734                signers_map,
735            },
736        );
737        if summary.content_digest != *contents.digest() {
738            return Err(anyhow!(
739                "checkpoint summary content digest does not match decoded contents: \
740                 summary={:?} contents={:?}",
741                summary.content_digest,
742                contents.digest(),
743            ));
744        }
745        // TODO: should we fetch the committee and pass that into try_into_verified instead of
746        // constructing this with new_unchecked?
747        Ok((VerifiedCheckpoint::new_unchecked(certified), contents))
748    }
749
750    fn decode_bcs<T>(field: Option<Base64>, label: &str) -> Result<T, Error>
751    where
752        T: serde::de::DeserializeOwned,
753    {
754        let bytes = FastCryptoBase64::decode(
755            &field
756                .ok_or_else(|| anyhow!("Missing {} in checkpoint response", label))?
757                .0,
758        )
759        .with_context(|| format!("{} does not decode", label))?;
760        bcs::from_bytes(&bytes).with_context(|| format!("cannot deserialize {}", label))
761    }

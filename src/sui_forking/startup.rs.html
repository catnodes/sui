28/// Initialize a forked network by fetching state from the remote endpoint at
29/// `forked_at_checkpoint` and starting a local Simulacrum instance.
30pub async fn initialize(
31    node: Node,
32    forked_at_checkpoint: CheckpointSequenceNumber,
33    version: &str,
34) -> Result<Context> {
35    // 1. Create DataStore — empty local cache, GraphQL wired up.
36    let data_store = DataStore::new(node.clone(), forked_at_checkpoint, version).await?;
37    let chain_identifier = data_store.chain();
38
39    // 2. Download and persist the startup checkpoint (summary + contents),
40    //    then read the summary back through the cache-aware getter.
41    data_store.download_and_persist_startup_checkpoint()?;
42    let checkpoint = data_store
43        .get_checkpoint_by_sequence_number(forked_at_checkpoint)?
44        .ok_or_else(|| anyhow!("checkpoint {} not found", forked_at_checkpoint))?;
45
46    // 3. Read system state — fetches object 0x5 + dynamic fields from remote via ObjectStore.
47    let system_state =
48        sui_types::sui_system_state::get_sui_system_state(&data_store).map_err(|e| {
49            anyhow!(
50                "failed to read system state at checkpoint {}: {}",
51                forked_at_checkpoint,
52                e
53            )
54        })?;
55
56    let protocol_version = system_state.protocol_version();
57
58    // 4. Build NetworkConfig with local test validators.
59    let mut rng = OsRng;
60    let config = ConfigBuilder::new_with_temp_dir()
61        .rng(&mut rng)
62        .with_chain_start_timestamp_ms(checkpoint.timestamp_ms)
63        .deterministic_committee_size(NonZeroUsize::MIN)
64        .with_protocol_version(ProtocolVersion::new(protocol_version))
65        .with_chain_override(node.chain())
66        .build();
67
68    // 5. Override validators in system state with local keys from config.
69    let system_state = override_validators(system_state, &config)?;
70
71    // 6. Build KeyStore.
72    let keystore = KeyStore::from_network_config(&config);
73
74    // 7. Create Simulacrum from custom state.
75    let simulacrum = Simulacrum::new_from_custom_state(
76        keystore,
77        checkpoint,
78        system_state,
79        &config,
80        data_store,
81        rng,
82    );
83
84    Ok(Context::new(simulacrum, chain_identifier))
85}

87/// Run the forked network. Spawns a `sui-rpc-api` gRPC server bound to
88/// `rpc_addr` on top of the context's `DataStore`, then blocks on Ctrl+C.
89pub async fn run(context: Context, rpc_addr: SocketAddr, version: &'static str) -> Result<()> {
90    let reader: Arc<dyn RpcStateReader> = {
91        let sim = context.simulacrum().read().await;
92        Arc::new(sim.store().clone())
93    };
94    let mut service = RpcService::new(reader);
95    service.with_server_version(ServerVersion::new("sui-forking", version));
96    service.with_executor(Arc::new(ForkedTransactionExecutor::new(context.into())));
97
98    info!("starting sui-rpc-api server on {rpc_addr}");
99    let server_handle = tokio::spawn(async move { service.start_service(rpc_addr).await });
100
101    info!("forked network running, waiting for shutdown signal (Ctrl+C)");
102    tokio::select! {
103        res = tokio::signal::ctrl_c() => {
104            res?;
105            info!("shutdown signal received, stopping forked network");
106        }
107        // If the RPC server task aborts unexpectedly, surface that as an
108        // error rather than blocking on Ctrl+C indefinitely.
109        join = server_handle => {
110            if let Err(e) = join {
111                return Err(anyhow!("rpc server task panicked: {e}"));
112            }
113            return Err(anyhow!("rpc server task exited unexpectedly"));
114        }
115    }
116    Ok(())
117}

119/// Replace the validator set in the system state with local validators from the NetworkConfig
120/// genesis, so the simulacrum can sign checkpoints with locally available keys.
121fn override_validators(
122    system_state: SuiSystemState,
123    config: &NetworkConfig,
124) -> Result<SuiSystemState> {
125    let genesis_validators = match config.genesis.sui_system_object() {
126        SuiSystemState::V1(inner) => inner.validators,
127        SuiSystemState::V2(inner) => inner.validators,
128        #[cfg(msim)]
129        _ => anyhow::bail!("unsupported genesis system state variant"),
130    };
131
132    match system_state {
133        SuiSystemState::V1(mut inner) => {
134            inner.validators = genesis_validators;
135            Ok(SuiSystemState::V1(inner))
136        }
137        SuiSystemState::V2(mut inner) => {
138            inner.validators = genesis_validators;
139            Ok(SuiSystemState::V2(inner))
140        }
141        #[cfg(msim)]
142        _ => anyhow::bail!("unsupported system state variant"),
143    }
144}
